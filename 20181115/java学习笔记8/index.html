同时在try 里面用bufferedinputstream 和bufferedoutputstream 导致错误!!!只能分开2个try写或者不用buffer

`java
class TestStream {
public static void main(String[] args) {
File f = new File(“f:\lol.txt”);
try (BufferedReader fr = new BufferedReader(new FileReader(f))) {
String line;
//错误写法：while ((String line =br.readLine()) !=null) {
while ((line = fr.readLine()) != null) { //while的妙用！
System.out.println(line);
}
} catch (IOException e) {
e.printStackTrace();
}
System.out.println(“文件已读取完毕!”);
}
}

class TestStreammmm {

public static void main(String[] args) {
    File f = new File("f:/lol.txt");
    try (BufferedWriter fr = new BufferedWriter(new FileWriter(f, true))) {        //append:true 添加模式！非覆盖模式
        String str = "43434\nre\n   \n998 999\n";
        fr.write(str);
    } catch (IOException e) {
        e.printStackTrace();
    }
}

}
